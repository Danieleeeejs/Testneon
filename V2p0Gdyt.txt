-- Animation function
function animateButton(button)
    local originalSize = button.Size
    local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    
    local scaleUp = TweenService:Create(button, tweenInfo, {
        Size = originalSize + UDim2.new(0, -5, 0, -5)
    })
    
    local scaleDown = TweenService:Create(button, tweenInfo, {
        Size = originalSize
    })
    
    scaleUp:Play()
    scaleUp.Completed:Connect(function()
        scaleDown:Play()
    end)
end

-- Attribute toggle functionality (optional)
local attributeStates = {
    Fly = false,
    Ride = false,
    Neon = false,
    MegaNeon = false
}

-- You can add functionality to toggle attributes on/off by clicking them
for i, attribute in pairs(attributeNames) do
    local attributeFrame = attributesFrame:FindFirstChild(attribute .. "Attribute")
    if attributeFrame then
        attributeFrame.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                attributeStates[attributeDescriptions[i]] = not attributeStates[attributeDescriptions[i]]
                
                local label = attributeFrame:FindFirstChild("Label")
                if attributeStates[attributeDescriptions[i]] then
                    label.TextColor3 = Color3.fromRGB(255, 255, 255)
                    attributeFrame.BackgroundColor3 = attributeColors[i]
                else
                    label.TextColor3 = attributeColors[i]
                    attributeFrame.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
                end
            end
        end)
    end
end
